import json
from channels.generic.websocket import AsyncWebsocketConsumer
from channels.db import database_sync_to_async
from .models import ChatMessage
from django.contrib.auth.models import User
from django.utils import timezone

class ChatConsumer(AsyncWebsocketConsumer):
    async def connect(self):
        self.room_group_name = "chat_global_chat"
        await self.channel_layer.group_add(self.room_group_name, self.channel_name)
        await self.accept()

    async def disconnect(self, close_code):
        await self.channel_layer.group_discard(self.room_group_name, self.channel_name)

    async def receive(self, text_data):
        try:
            data = json.loads(text_data)
        except json.JSONDecodeError:
            return

        msg_type = data.get('msg_type')
        username = data.get('username')

        # RTC SIGNALING (WebRTC Offer/Answer/Candidate)
        if msg_type in ['rtc_offer', 'rtc_answer', 'rtc_candidate', 'rtc_hangup']:
            await self.channel_layer.group_send(
                self.room_group_name,
                {
                    'type': 'chat_message',
                    'sender_channel_name': self.channel_name, 
                    **data 
                }
            )
            return 

        # NORMAL CHAT LOGIC
        success = await self.save_message(data)
        
        if success:
            now_local = timezone.localtime(timezone.now())
            await self.channel_layer.group_send(
                self.room_group_name,
                {
                    'type': 'chat_message',
                    'sender_channel_name': self.channel_name,
                    'message': data.get('message'),
                    'username': username,
                    'msg_type': msg_type,
                    'media_url': data.get('media_url', ''),
                    'timestamp': f"Today, {now_local.strftime('%H:%M:%S')}"
                }
            )
    
    @database_sync_to_async
    def save_message(self, data):
        username = data.get('username')
        if not username: return False
        try:
            user = User.objects.get(username=username)
            ChatMessage.objects.create(
                user=user,
                message=data.get('message', ''),
                msg_type=data.get('msg_type', 'text'),
                media_url=data.get('media_url', '')
            )
            return True
        except User.DoesNotExist:
            return False

    async def chat_message(self, event):
        # Don't send back to sender
        if self.channel_name == event.get('sender_channel_name'):
            return 

        client_payload = {k: v for k, v in event.items() if k not in ['type', 'sender_channel_name']}
        await self.send(text_data=json.dumps(client_payload))

def home_view(request):
    return JsonResponse({"message": "Backend is running!", "status": "OK"})

@api_view(['GET', 'POST'])
@permission_classes([AllowAny])
def register_view(request):
    if request.method == 'GET':
        return Response({"message": "Send a POST request with username and password to register."})

    username = request.data.get('username')
    password = request.data.get('password')
    
    if not username or not password:
        return Response({"error": "Username and password required"}, status=status.HTTP_400_BAD_REQUEST)
    
    if User.objects.filter(username=username).exists():
        return Response({"error": "Username already exists"}, status=status.HTTP_400_BAD_REQUEST)
    
    try:
        User.objects.create_user(username=username, password=password)
        return Response({"message": "User created successfully"}, status=status.HTTP_201_CREATED)
    except Exception as e:
        return Response({"error": str(e)}, status=status.HTTP_500_INTERNAL_SERVER_ERROR)

@api_view(['GET'])
@permission_classes([IsAuthenticated])
def protected_view(request):
    return Response({"message": f"Hello {request.user.username}!", "user_id": request.user.id})

@api_view(['POST'])
@permission_classes([IsAuthenticated])
@parser_classes([MultiPartParser, FormParser])
def upload_file_view(request):
    if 'file' not in request.FILES:
        return Response({"error": "No file provided"}, status=status.HTTP_400_BAD_REQUEST)
    
    file_obj = request.FILES['file']
    file_name = default_storage.save(f"uploads/{file_obj.name}", file_obj)
    file_url = default_storage.url(file_name)
    
    # Construct URL for frontend
    # Note: For local HTTP, http://127.0.0.1:8000/media/... works
    full_url = request.build_absolute_uri(file_url)
    
    return Response({"file": full_url}, status=status.HTTP_201_CREATED)

@api_view(['GET'])
@permission_classes([IsAuthenticated])
def get_chat_history(request):
    messages = ChatMessage.objects.all().order_by('-timestamp')[:50]
    
    now_local = timezone.localtime(timezone.now())
    today = now_local.date()
    yesterday = today - timedelta(days=1)
    
    data = []
    for m in reversed(messages):
        local_datetime = timezone.localtime(m.timestamp)
        msg_date = local_datetime.date()
        msg_time = local_datetime.strftime("%H:%M")

        if msg_date == today:
            display_timestamp = f"Today, {msg_time}"
        elif msg_date == yesterday:
            display_timestamp = f"Yesterday, {msg_time}"
        else:
            display_timestamp = local_datetime.strftime("%d %b, %H:%M")
        
        data.append({
            "username": m.user.username,
            "message": m.message,
            "msg_type": m.msg_type,
            "media_url": m.media_url, 
            "timestamp": display_timestamp 
        })
    return Response(data)

    const handleAuth = async (e) => {
        e.preventDefault();

                onLoginSuccess(data.access, username);

                setError(data.detail || "Invalid credentials");

        } catch (err) {
            setError("Network Error");
        }

        <div className="auth-container">
            <div className="auth-card">

                </form>

    );

import React, { useState, useEffect } from "react";
import Login from "./components/Login";
import Register from "./components/Register";
import Chat from "./components/Chat";
import "./App.css";

function App() {
  const [token, setToken] = useState("");
  const [username, setUsername] = useState("");
  const [view, setView] = useState("login"); // 'login', 'register', 'chat'

  useEffect(() => {
    const savedToken = localStorage.getItem("token");
    const savedUser = localStorage.getItem("username");
    if (savedToken) {
      setToken(savedToken);
      setUsername(savedUser || "");
      setView("chat");
    }
  }, []);

  const handleLoginSuccess = (newToken, user) => {
    setToken(newToken);
    setUsername(user);
    localStorage.setItem("token", newToken);
    localStorage.setItem("username", user);
    setView("chat");
  };

  const handleRegisterSuccess = () => {
    alert("Registration successful! Please login.");
    setView("login");
  };

  const handleLogout = () => {
    setToken("");
    setUsername("");
    localStorage.removeItem("token");
    localStorage.removeItem("username");
    setView("login");
  };

  return (
    <div className="App">
      {view === "login" && (
        <Login onLoginSuccess={handleLoginSuccess} onSwitchToRegister={() => setView("register")} />
      )}
      {view === "register" && (
        <Register onRegisterSuccess={handleRegisterSuccess} onSwitchToLogin={() => setView("login")} />
      )}
      {view === "chat" && (
        <Chat token={token} username={username} onLogout={handleLogout} />
      )}
    </div>
  );
}

export default App;

    const { sendJsonMessage, lastMessage, readyState } = useWebSocket(WS_URL, {

        return () => document.removeEventListener('click', handleClickOutside);
    }, []);

    }, [readyState]);

    // Cleanup camera on unmount
    useEffect(() => {
        return () => stopCall();
    }, []);

    // Scroll to bottom
    useEffect(() => {
        messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
    }, [chatHistory]);

    // Load History
    useEffect(() => {

                headers: { Authorization: `Bearer ${token}` }
            })
                .then(res => {

                        _tempId: `hist_${index}`
                    }));

                })
                .catch(err => console.error(err));
        }
    }, [token]);

    // Handle Socket Messages (Robust Parsing)
    useEffect(() => {
        if (!lastMessage) return;

        let dataStr = "";
        if (typeof lastMessage === 'object' && lastMessage.data) {
            dataStr = lastMessage.data;
        } else if (typeof lastMessage === 'string') {
            dataStr = lastMessage;
        } else {
            return;
        }

        if (!dataStr || !dataStr.startsWith('{')) return;

        try {
            const data = JSON.parse(dataStr);
            const { msg_type, offer, answer, candidate, username: sender, call_mode, _tempId } = data;

            // Handle Sync Actions (Edit/Delete/React)
            if (msg_type === 'delete_message') {
                setChatHistory(prev => prev.map(m => 
                    (m._tempId === data.targetId || (m.timestamp === data.timestamp && m.username === sender))
                    ? { ...m, message: "This message was deleted", msg_type: 'system', isDeleted: true }

                return;
            }
            if (msg_type === 'edit_message') {

                    ? { ...m, message: data.newMessage }
                    : m

            }
            if (msg_type === 'add_reaction') {
                setChatHistory(prev => prev.map(m => 
                    (m._tempId === data.targetId || (m.timestamp === data.timestamp && m.username === sender))
                    ? { ...m, reactions: [...(m.reactions || []), { user: sender, emoji: data.emoji }] }
                    : m
                ));
                return;
            }

            if (sender === username) return;

            // 1. WebRTC Signaling
            if (msg_type === 'rtc_offer') {
                if (window.confirm(`${sender} is calling (${call_mode}). Accept?`)) {
                    startLocalStream(call_mode === 'video').then(async (success) => {
                        if (success && pc.current) {
                            setCallType(call_mode);
                            setIsCalling(false);
                            try {
                                await pc.current.setRemoteDescription(new RTCSessionDescription(offer));
                                const ans = await pc.current.createAnswer();
                                await pc.current.setLocalDescription(ans);
                                sendJsonMessage({ msg_type: 'rtc_answer', answer: ans, username });
                            } catch (err) { console.error(err); }
                        }
                    });
                }
            }
            else if (msg_type === 'rtc_answer') {
                setIsCalling(false);
                if (pc.current) {
                    pc.current.setRemoteDescription(new RTCSessionDescription(answer));
                }
            }
            else if (msg_type === 'rtc_candidate') {
                if (pc.current) {
                    pc.current.addIceCandidate(new RTCIceCandidate(candidate));
                }
            }
            else if (msg_type === 'rtc_hangup') {
                addSystemMessage(`${sender} ended the call`);
                stopCall();
            }
            // 2. Chat Messages
            else {
                setTimeout(() => {
                    setChatHistory(prev => [...prev, { ...data, _tempId: `msg_${Date.now()}` }]);
                }, 0);
            }
        } catch (error) {
            console.error("JSON Parse Error:", error);
        }
    }, [lastMessage, username, sendJsonMessage]);

    const addSystemMessage = (text) => {
        setTimeout(() => {
            setChatHistory(prev => [...prev, {
                username: 'System',
                message: text,
                msg_type: 'system',
                timestamp: 'Just now'
            }]);
        }, 0);
    };

    // --- NEW: ACTION HANDLERS ---
    const handleDelete = (index, msg) => {
        // 1. Update Local State
        const updatedHistory = [...chatHistory];
        updatedHistory[index] = { ...updatedHistory[index], message: "This message was deleted", msg_type: 'system', isDeleted: true };
        setChatHistory(updatedHistory);
        setActiveMenuIndex(null);

        // 2. Send Signal
        sendJsonMessage({
            msg_type: 'delete_message',
            username: username,
            targetId: msg._tempId,
            timestamp: msg.timestamp
        });
    };

    const handleEditStart = (index, msg) => {
        setEditingIndex(index);
        setEditText(msg.message);
        setActiveMenuIndex(null);
    };

    const handleEditSave = (index, msg) => {
        // 1. Update Local State
        const updatedHistory = [...chatHistory];
        updatedHistory[index] = { ...updatedHistory[index], message: editText };
        setChatHistory(updatedHistory);
        setEditingIndex(null);

        // 2. Send Signal
        sendJsonMessage({
            msg_type: 'edit_message',
            username: username,
            newMessage: editText,
            targetId: msg._tempId,
            timestamp: msg.timestamp
        });
    };

    const handleReact = (index, msg, emoji) => {
        // 1. Update Local State
        const updatedHistory = [...chatHistory];
        const reactions = updatedHistory[index].reactions || [];
        // Prevent duplicate reaction from same user
        if (!reactions.find(r => r.user === username && r.emoji === emoji)) {
            updatedHistory[index] = { ...updatedHistory[index], reactions: [...reactions, { user: username, emoji }] };
            setChatHistory(updatedHistory);

            // 2. Send Signal
            sendJsonMessage({
                msg_type: 'add_reaction',
                username: username,
                emoji: emoji,
                targetId: msg._tempId,
                timestamp: msg.timestamp
            });
        }
        setReactingIndex(null);
        setActiveMenuIndex(null);
    };


    // --- WebRTC Logic ---
    const startLocalStream = async (isVideo) => {
        try {
            const stream = await navigator.mediaDevices.getUserMedia({ video: isVideo, audio: true });
            setLocalStream(stream);
            if (localVideoRef.current) localVideoRef.current.srcObject = stream;

            pc.current = new RTCPeerConnection({ iceServers: [{ urls: "stun:stun.l.google.com:19302" }] });

            stream.getTracks().forEach(track => pc.current.addTrack(track, stream));

            pc.current.onicecandidate = (e) => {
                if (e.candidate) sendJsonMessage({ msg_type: 'rtc_candidate', candidate: e.candidate, username });
            };

            pc.current.ontrack = (e) => {
                if (remoteVideoRef.current) remoteVideoRef.current.srcObject = e.streams[0];
            };
            return true;
        } catch (err) {
            console.error("Media Error:", err);
            alert("Could not access media devices.");
            return false;
        }
    };

    const handleCall = async (mode) => {
        const success = await startLocalStream(mode === 'video');
        if (success && pc.current) {
            setCallType(mode);
            setIsCalling(true);
            const offer = await pc.current.createOffer();
            await pc.current.setLocalDescription(offer);
            sendJsonMessage({ msg_type: 'rtc_offer', offer, username, call_mode: mode });
        }
    };

    const stopCall = () => {
        if (pc.current) { pc.current.close(); pc.current = null; }
        if (localStream) { localStream.getTracks().forEach(t => t.stop()); setLocalStream(null); }
        setCallType(null);
        setIsCalling(false);
        setIsMicMuted(false);
        setIsVideoMuted(false);
    };

    // --- Mute/Unmute Logic ---
    const toggleAudio = () => {
        if (!localStream) return;
        const audioTrack = localStream.getAudioTracks()[0];
        if (audioTrack) {
            audioTrack.enabled = !audioTrack.enabled;
            setIsMicMuted(!audioTrack.enabled);
        }
    };

    const toggleVideo = () => {
        if (!localStream) return;
        const videoTrack = localStream.getVideoTracks()[0];
        if (videoTrack) {
            videoTrack.enabled = !videoTrack.enabled;
            setIsVideoMuted(!videoTrack.enabled);
        }
    };

    // --- Chat Logic ---
    const handleSendMessage = () => {
        if (message.trim()) {
            const msgData = { message, username, msg_type: 'text', timestamp: 'Just now', _tempId: `msg_${Date.now()}` };
            setTimeout(() => setChatHistory(prev => [...prev, msgData]), 0);
            sendJsonMessage(msgData);
            setMessage("");
            setShowEmoji(false);
        }
    };

    const handleFileUpload = async (e) => {
        const file = e.target.files[0];
        if (!file) return;
        await uploadFile(file);
        e.target.value = null;
    };

    // --- Audio Recording ---
    const toggleRecording = async () => {
        if (isRecording) stopRecording();
        else await startRecording();
    };

    const startRecording = async () => {
        try {
            const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
            mediaRecorderRef.current = new MediaRecorder(stream);
            audioChunksRef.current = [];

            mediaRecorderRef.current.ondataavailable = (e) => {
                if (e.data.size > 0) audioChunksRef.current.push(e.data);
            };

            mediaRecorderRef.current.onstop = () => {
                const blob = new Blob(audioChunksRef.current, { type: 'audio/webm' });
                const file = new File([blob], `voice_${Date.now()}.webm`, { type: 'audio/webm' });
                uploadFile(file);
                stream.getTracks().forEach(t => t.stop());
            };

            mediaRecorderRef.current.start();
            setIsRecording(true);
        } catch (err) {
            alert("Mic access denied.");
        }
    };

    const stopRecording = () => {
        if (mediaRecorderRef.current && mediaRecorderRef.current.state !== 'inactive') {
            mediaRecorderRef.current.stop();
            setIsRecording(false);
        }
    };

    const uploadFile = async (file) => {
        const type = file.type.split('/')[0];
        const formData = new FormData();
        formData.append('file', file);

        try {
            const res = await axios.post(`${BACKEND_URL}/api/upload/`, formData, {
                headers: { 'Authorization': `Bearer ${token}` }
            });

            const msgData = {
                username,
                msg_type: ['image', 'video', 'audio'].includes(type) ? type : 'file',
                media_url: res.data.file,
                message: `Sent a ${type}`,
                timestamp: 'Just now',
                _tempId: `msg_${Date.now()}`
            };

            setTimeout(() => setChatHistory(prev => [...prev, msgData]), 0);
            sendJsonMessage(msgData);

        } catch (err) { alert("Upload failed"); }
    };

    const renderMessage = (msg) => {
        if (msg.isDeleted) return <span style={{ fontStyle: 'italic', color: '#888' }}>{msg.message}</span>;

        const fullUrl = msg.media_url?.startsWith('http') ? msg.media_url : `${BACKEND_URL}${msg.media_url}`;
        
        if (msg.username === 'System') {
            return <div className="system-msg">{msg.message}</div>;
        }

        if (msg.msg_type === 'image') return <img src={fullUrl} className="chat-media" alt="sent" />;
        if (msg.msg_type === 'video') return <video src={fullUrl} controls className="chat-media" />;
        if (msg.msg_type === 'audio') return <audio src={fullUrl} controls className="chat-audio" />;
        if (msg.msg_type === 'file') return <a href={fullUrl} target="_blank" rel="noreferrer">📎 File</a>;
        return <span>{msg.message}</span>;
    };

    return (
        <div className="chat-container">
            <div className="chat-header">
                <h2>Global Chat ({username})</h2>
                <div className="header-actions">
                    <button onClick={() => handleCall('audio')} disabled={!!localStream}>📞</button>
                    <button onClick={() => handleCall('video')} disabled={!!localStream}>🎥</button>
                    <button onClick={onLogout} className="logout-btn">Logout</button>
                </div>
            </div>

            {/* Video Overlay */}
            {(localStream || isCalling) && (
                <div className="video-overlay">
                    {isCalling && <h3>Calling... ({callType})</h3>}
                    <div className="video-grid">
                        {callType === 'video' && <video ref={localVideoRef} autoPlay muted playsInline className="local-video" />}
                        <video ref={remoteVideoRef} autoPlay playsInline className="remote-video" />
                    </div>
                    <div className="call-controls">
                        <button onClick={toggleAudio} className={`control-btn ${isMicMuted ? 'off' : ''}`}>
                            {isMicMuted ? '🔇' : '🎤'}
                        </button>
                        {callType === 'video' && (
                            <button onClick={toggleVideo} className={`control-btn ${isVideoMuted ? 'off' : ''}`}>
                                {isVideoMuted ? '🚫' : '🎥'}
                            </button>
                        )}
                        <button 
                            onClick={() => {
                                addSystemMessage("You ended the call");
                                sendJsonMessage({ msg_type: 'rtc_hangup', username });
                                stopCall();
                            }} 
                            className="end-call-btn"
                        >
                            End Call
                        </button>
                    </div>
                </div>
            )}

            <div className="chat-box" onClick={() => setActiveMenuIndex(null)}>
                {chatHistory.map((msg, i) => (
                    <div key={msg._tempId || i} className={`message-row ${msg.username === username ? 'sent' : 'received'}`}>
                        {msg.username !== username && msg.username !== 'System' && <div className="sender-name">{msg.username}</div>}
                        
                        {/* Message Container */}
                        <div className="message-bubble-container">
                            {/* Reaction List */}
                            {msg.reactions && msg.reactions.length > 0 && (
                                <div className="reactions-list">
                                    {msg.reactions.map((r, idx) => (
                                        <span key={idx} className="reaction-bubble">{r.emoji}</span>
                                    ))}
                                </div>
                            )}

                            {/* Bubble Wrapper */}
                            <div 
                                className={`message-bubble ${activeMenuIndex === i ? 'selected' : ''}`}
                                style={msg.username === 'System' ? { background: 'transparent', color: '#888', padding: '5px' } : {}}
                                onClick={(e) => {
                                    if(msg.username === username && !msg.isDeleted) {
                                        e.stopPropagation(); // Stop bubble from closing
                                        setActiveMenuIndex(activeMenuIndex === i ? null : i);
                                    }
                                }}
                            >
                                {/* Render Content or Edit Input */}
                                {editingIndex === i ? (
                                    <div style={{ display: 'flex', gap: '5px' }}>
                                        <input 
                                            autoFocus
                                            value={editText} 
                                            onChange={e => setEditText(e.target.value)}
                                            onKeyDown={e => e.key === 'Enter' && handleEditSave(i, msg)}
                                            onClick={e => e.stopPropagation()}
                                            style={{ width: '100%', padding: '5px', borderRadius: '4px', border: '1px solid #ccc' }}
                                        />
                                        <button onClick={() => handleEditSave(i, msg)} style={{ cursor: 'pointer' }}>✅</button>
                                    </div>
                                ) : (
                                    <>
                                        {renderMessage(msg)}
                                        <div className="msg-time">{msg.timestamp}</div>
                                    </>
                                )}
                            </div>

                            {/* Context Menu (Edit/Delete/React) */}
                            {activeMenuIndex === i && (
                                <div className="message-menu" onClick={e => e.stopPropagation()}>
                                    <button onClick={() => setReactingIndex(i)}>React 👍</button>
                                    {msg.username === username && (
                                        <>
                                            <button onClick={() => handleEditStart(i, msg)}>Edit ✎️</button>
                                            <button onClick={() => handleDelete(i, msg)} style={{ color: 'red' }}>Delete 🗑️</button>
                                        </>
                                    )}
                                </div>
                            )}

                            {/* Emoji Picker for Reactions */}
                            {reactingIndex === i && (
                                <div className="reaction-picker-popup" onClick={e => e.stopPropagation()}>
                                    {['❤️', '👍', '😂', '😮', '😢', '😡'].map(emoji => (
                                        <span key={emoji} onClick={() => handleReact(i, msg, emoji)}>{emoji}</span>
                                    ))}
                                </div>
                            )}
                        </div>
                    </div>
                ))}
                <div ref={messagesEndRef} />
            </div>

            {showEmoji && <div className="emoji-picker"><EmojiPicker onEmojiClick={(e) => setMessage(prev => prev + e.emoji)} /></div>}

            <div className="input-area">
                <button onClick={() => setShowEmoji(!showEmoji)} className="icon-btn">😊</button>
                <button onClick={() => fileInputRef.current.click()} className="icon-btn">📎</button>
                <input type="file" ref={fileInputRef} onChange={handleFileUpload} hidden accept="image/*,video/*,audio/*,.pdf" />

                <button onClick={toggleRecording} className={`icon-btn mic-btn ${isRecording ? 'recording' : ''}`}>
                    {isRecording ? '⏹️' : '🎤'}
                </button>

                <input
                    value={message}
                    onChange={e => setMessage(e.target.value)}
                    onKeyDown={e => e.key === 'Enter' && handleSendMessage()}
                    placeholder="Type..."
                />
                <button onClick={handleSendMessage} className="send-btn">Send</button>
            </div>
        </div>
    );
};

export default Chat; this is chat.jsx file /
